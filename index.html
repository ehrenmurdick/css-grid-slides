type Msg
    = Increment
    | Decrement

type alias Model =
    { number : Int }

init : Model
init =
    { number = 0 }



update : Msg -> Model -> Model
update msg model =
    case msg of
        Increment ->
            { model | number = model.number + 1 }

        Decrement ->
            { model | number = model.number - 1 }


view : Model -> Html Msg
view model =
    div []
        [ button [ onClick Increment ] [ text "+" ]
        , text (String.fromInt model.number)
        , button [ onClick Decrement ] [ text "-" ]
        ]
